 startCounter() {
    // Ye code Angular ke zone ke bahar chalega
    this.ngZone.runOutsideAngular(() => {
      let i = 0;
      const interval = setInterval(() => {
        i++;
        console.log('Counter:', i);

        // Jab counter 5 tak pohche, Angular ke zone mein run kar ke UI update karenge
        if (i === 5) {
          clearInterval(interval);
          this.ngZone.run(() => {
            this.counter = i; // Ab Angular UI update karega
          });
        }
      }, 1000);
    });
  }
